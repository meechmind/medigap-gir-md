<div class="gir-triggers">
    <h2>Plan Problems Triggers</h2>
    <div style="margin-top: 1.5rem; max-width: 900px;">
        <div class="trigger-item">
            <div class="trigger-number">2</div>
            <div class="trigger-text">
                <strong>Medicare Advantage Plan Issues (Involuntary)</strong>
                <ul style="margin-top: 0.5rem; font-size: 0.9rem;">
                    <li>Plan terminated or decertified by CMS</li>
                    <li>Plan exits your geographic area</li>
                    <li>You move outside the service area</li>
                </ul>
                <span class="citation">COMAR 31.10.06.09-1(B)(2)(a)-(c)<sup>13</sup></span>
            </div>
        </div>
        <div class="trigger-item">
            <div class="trigger-number">3</div>
            <div class="trigger-text">
                <strong>Medicare Advantage Plan Misconduct</strong>
                <ul style="margin-top: 0.5rem; font-size: 0.9rem;">
                    <li>Plan substantially violated its contract with CMS</li>
                    <li>Plan materially misrepresented provisions when marketing</li>
                </ul>
                <span class="citation">COMAR 31.10.06.09-1(B)(2)(d)<sup>14</sup></span>
            </div>
        </div>
        </div>
    <div class="warning-box" style="margin-top: 1rem;">
        <strong>Key Distinction:</strong> Trigger 3 requires demonstrating plan wrongdoing, not just dissatisfaction.
    </div>
</div>
